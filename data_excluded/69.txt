#include <stdio.h>
#include <stdbool.h>

void read_matrix(int *matrix)
{
  for (int i = 0; i < 81; i++) {
    scanf("%d", &matrix[i]);
  }
}

void print_matrix(int *matrix)
{
  for (int i = 0; i < 81; i++) {
    printf("%d ", matrix[i]);
    if ((i + 1) % 9 == 0)
      printf("\n");
  }
}

bool find_unassigned(int *matrix, int *row, int *col)
{
  for (int i = 0; i < 9; i++) {
    for (int j = 0; j < 9; j++) {
      if (matrix[i * 9 + j] == 0) {
        *row = i;
        *col = j;
        return true;
      }
    }
  }
  return false;
}

bool no_conflicts(int *matrix, int row, int col, int n)
{
  for (int i = 0; i < 9; i++) {
    if (matrix[row * 9 + i] == n || matrix[i * 9 + col] == n)
      return false;
  }
  for (int i = row - row % 3; i < row - row % 3 + 3; i++) {
    for (int j = col - col % 3; j < col - col % 3 + 3; j++) {
      if (matrix[i * 9 + j] == n)
        return false;
    }
  }

  return true;
}

bool solve_matrix(int *matrix)
{
  int row, col;

  if (!find_unassigned(matrix, &row, &col))
    return true;

  for (int n = 1; n <= 9; n++) {
    if (no_conflicts(matrix, row, col, n)) {
      matrix[row * 9 + col] = n;
      if (solve_matrix(matrix))
        return true;
      matrix[row * 9 + col] = 0;
    }
  }

  return false;
}

int main(void)
{
  int matrix[81];

  read_matrix(matrix);

  if (solve_matrix(matrix))
    print_matrix(matrix);
  else
    printf("%s", "keine Lösung gefunden");

  return 0;
}
