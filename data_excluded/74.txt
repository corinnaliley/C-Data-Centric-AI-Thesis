int strendX(const char *s, const char *t) {  
  int laenge_s = strlen(s);
  int laenge_t = strlen(t);
  if (laenge_t > laenge_s) {
    return 0;
    }  
  char *p = s + laenge_s-1;
  char *q = t + laenge_t-1;  
  while (q >= t) {
    if (*q-- != *p--) { 
      return 0;
    }
  }  
  return 1; 
}

char *strchrX(const char *s, int c) {
  int laenge_s = strlen(s);
  char *p = s + laenge_s-1;
  while (p >= s) {
    if (c == *p) {
      return p;
    }
    p--;
  }
  return NULL;  
}

char *strstrX(const char *s, const char *t) {
  int laenge_s = strlen(s);
  int laenge_t = strlen(t); 
  
  char *end_s = s + laenge_s;
  char *copy_s = s;
  char *copy_t = t;
  
  char *p = s;
  char *q = t;
  
  if (laenge_t == 0) {
    return s;
  }
    
  while (copy_s < end_s) {
    if (*copy_s == *copy_t) {
      while(laenge_t--) {
        if (*copy_s++ != *copy_t++) {
          return NULL;
        }
      } 
      if (laenge_t == -1) {
        return p;
      }
    }
    copy_s++;
    p++;
  }
  return NULL;
}
