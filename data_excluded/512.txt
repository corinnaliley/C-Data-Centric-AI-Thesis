TASK:
Programmieren Sie eine Funktion `void simple_file_math(char *path)`, die einen String mit dem Pfad `path` zu einer Datei übergeben bekommt. 
In der Datei stehen zeilenweise kleine Rechenaufgaben, die gelöst und in die Datei geschrieben werden sollen.

Es gibt 3 Aufgabentypen:

- `quadrat <n>`: Das Quadrat der Zahl `n` soll berechnet werden.
- `mal2 <n>`: Die Zahl `n` soll mit 2 multipliziert werden.
- `plus` <n> <m>: Die zwei Zahlen `n` und `m` sollen aufaddiert werden.

In der Datei soll in jeder Zeile ihre Anweisung durch das Ergebnis der Rechnung ersetzt werden. Der gesamte Inhalt der Datei soll am Ende der Funktion geändert worden sein. 


### Beispielaufruf
```
beispiel.txt: 
  quadrat 6
  mal2 15
  plus 5 9
  quadrat 5

simple_file_math("beispiel.txt");
```
### Beispielergebnis
```
beispiel.txt: 
  36
  30
  14
  25  
```

void simple_file_math(char *path) {

  size_t len = 0;
  char delim[] = " ";
  char result[1000];
  if (fp == NULL) {
    printf("Error on opening file");
    return;
  }
  int num;
  int start = 0;
  while ((read = getline(&line, &len, fp)) != -1) {
    char *token = strtok(line, delim);
    int n = atoi(strtok(NULL, delim));
    char tmp[100];
    if (strcmp(token, "quadrat") == 0) {
      sprintf(tmp, "%d", n * n);
    }
    if (strcmp(token, "mal2") == 0) {
      sprintf(tmp, "%d", n * 2);
    }
    if (strcmp(token, "plus") == 0) {
      int m = atoi(strtok(NULL, delim));
      sprintf(tmp, "%d", n + m);
    }
    if (start != 0) {
      char newline[] = "\n";
      strcat(result, newline);
    }
    start++;
    strcat(result, tmp);
  }
  char newline[] = " \n";
  strcat(result, newline);
  fclose(fp);
  if (line)
    free(line);
  fp = fopen(path, "w");
  if (fp == NULL) {
    printf("Error on opening file");
    return;
  }
  fprintf(fp, "%s", result);
  fclose(fp);
}
