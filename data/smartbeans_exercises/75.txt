char *strinv(const char *s) {
    long laenge = strlen(s);
    char *ergebnis = (char*) malloc((laenge+1)*sizeof(char));
    for(int i = (laenge-1); i >= 0; i--) {
        *ergebnis = *(s+i);
        ergebnis++;
    }
    *ergebnis = '\0';

    return (ergebnis-laenge);
}

char *strconcat(const char *s, const char *t) {
    long laenge = strlen(s) + strlen(t);
    char* ergebnis = (char*) malloc((laenge+1)*sizeof(char));
    while(*s != '\0') {
        *ergebnis = *s;
        ergebnis++;
        s++;
    }
    while(*t != '\0') {
        *ergebnis = *t;
        ergebnis++;
        t++;
    }
    *ergebnis = '\0';

    return (ergebnis-laenge) ;
}
