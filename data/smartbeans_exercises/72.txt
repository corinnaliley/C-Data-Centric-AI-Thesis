int datum2int(int tag, int monat){
  int day=tag;
  for(int i=1; i<monat;i++){
    //printf("day: %d \n", day);
    if(i==2){
      day+=28;
    }
    else if((i%2)==0 && i<=7){
      day+=30;
    }
    else if((i%2)==0 && i>7){
      day+=31;
    }
    else if((i%2)==1 && i<=7){
      day+=31;
    }
    else{
      day+=30;
    }
  }
  return day;

}

void int2datum(int tage, int* ergebnis){
  int counter=tage;
  int *monthdays = (int *) calloc(13, sizeof(int));
  for(int i=1; i<=12;i++){
    if(i==2){
      monthdays[i]=28;
    }
    else if((i%2)==0 && i<=7){
      monthdays[i]=30;
    }
    else if((i%2)==0 && i>7){
      monthdays[i]=31;
    }
    else if((i%2)==1 && i<=7){
      monthdays[i]=31;
    }
    else{
      monthdays[i]=30;
    }
  }
  for(int i=1; i<=12;i++){
    //printf("counter: %d, monatstage: %d Datum: %d.%d \n", counter,monthdays[i], *(ergebnis+1),*ergebnis);
    if(counter-monthdays[i]<=0){
        *(ergebnis+1)=*(ergebnis+1)+1;
        *ergebnis=counter;
        break;
    }
    counter-=monthdays[i];
    *(ergebnis+1)=*(ergebnis+1)+1;

  }

  free(monthdays);
  return;
}
