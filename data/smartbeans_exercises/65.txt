#include <stdio.h>
 
int main(){
  int jahr;
  scanf("%d", &jahr);
  switch(jahr % 4){
    case 0: printf("Sommerspiele");
            break;
    case 2: printf("Winterspiele");
            break;
    default: printf("keine Spiele");
             break;
  }
}
