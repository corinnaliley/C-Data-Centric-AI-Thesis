#include <string.h>
 
int exchange(char* s, char a, char b){
  int len = strlen(s);
  int count = 0;
  for(int i = 0; i < len; i++){
    if(s[i] == a){
      s[i] = b;
      count++;
    }
  }
  return count;
}
