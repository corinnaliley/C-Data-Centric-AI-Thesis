TASK:
Eine CSV Datei ist eine spezielle Textdatei für das Speichern von strukturierten Daten. Auf Wikipedia können Sie sich [hier](https://de.wikipedia.org/wiki/CSV_(Dateiformat)) mit dem genauen Aufbau vertraut machen.

Schreiben Sie eine Funktion 
```
char ***parse_csv(char *csv)
``` 
die den Inhalt einer CSV Datei als String übergeben bekommt und diesen parst. 

*Anforderungen*

- Der CSV String soll dabei als zweidimensional Feld verstanden werden. 
- Die verschiedenen Zeilen sind durch `\n` getrennt. Auch hinter der letzte Zeile folgt ein `\n`. 
- Die Einträge innerhalb der Zeilen sind durch das Trennzeichen `;` getrennt. 
- Der String enthält immer mind. ein Element.

Über den zurückgelieferten Pointer soll auf jedes Element aus dem String einzeln wie bei einem Feld zugegriffen werden können.

### Beispielaufruf
```
char * csv_string = "Ein;Beispiel;für\ndas;bessere;Verständnis\nder;Aufgabe;N5\n";
char *** csv_array = parse_csv(csv_string);
printf("%s\n", csv_array[0][0]);
printf("%s\n", csv_array[1][2]);
printf("%s\n", csv_array[2][1]);
```
### Beispielergebnis
```
Ein
Verständnis
Aufgabe
```

#include <stdlib.h>

#define DELIMITER ';'
#define ENDLINE '\n'

char ***parse_csv(char *csv) {
  int lines = 0;
  int columns = 1;

  char *p = csv;

  while (*p) {
    if (lines == 0 && *p == DELIMITER) {
      columns++;
    } else if (*p == ENDLINE) {
      lines++;
    }
    p++;
  }

  // Allocate memory to save all csv data
  char *data = (char *)malloc((size_t)((p - csv) * sizeof(char)));
  if (data == NULL) {
    return NULL;
  }

  // Allocate memory to hold line pointers
  char ***line_array = (char ***)malloc((size_t)(lines * sizeof(char **)));
  if (line_array == NULL) {
    return NULL;
  }

  for (int line = 0; line < lines; line++) {
    // Allocate memory to hold cell pointers
    char **cell_array = (char **)malloc((size_t)(columns * sizeof(char *)));
    if (cell_array == NULL) {
      return NULL;
    }

    line_array[line] = cell_array;
  }

  // As a last step, iterate over the csv data and do the following..
  // .. 1. Copy the csv content to the data memory
  // .. 2. Assign the corresponding pointers to each entry

  p = csv;
  char *d = data;
  line_array[0][0] = d;
  int line = 0;
  int column = 0;
  int end = 0;
  while (*p) {
    if (end == 1) {
      return NULL;
    }
    if (*p == DELIMITER) {
      if (column < columns) {
        column++;
        line_array[line][column] = d + 1;
        *d = '\0';
      } else {
        return NULL;
      }
    } else if (*p == ENDLINE) {
      column = 0;
      line++;
      if (line < lines) {
        line_array[line][column] = d + 1;
        *d = '\0';
      } else {
        end = 1;
      }
    } else {
      *d = *p;
    }
    d++;
    p++;
  }

  return line_array;
}
