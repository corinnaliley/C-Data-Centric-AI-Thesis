#include<stdio.h>
  
  #define MAXLEN 15
  
  int error() {
    printf("Fehler\n");
    return 1;
  }
  int main() {
    char str1[MAXLEN] = {0};
    char str2[MAXLEN] = {0};
    int str1_length = 0;
    int str2_length = 0;
    while(1) {
      int c = getchar();
      if(c == EOF)
        return error();
      if(c == '\n')
        break;
      if(str1_length == MAXLEN-1)
        return error();
      str1[str1_length++] = c;
    }
    while(1) {
      int c = getchar();
      if(c == EOF || c == '\n')
        break;
      if(str2_length == MAXLEN-1)
        return error();
      str2[str2_length++] = c;
    }
    printf("%s%s", str2, str1);
  }
